CS140 -- Midterm Exam: Q8 - Q11

The following code is for questions 8 through 11 
#include "fields.h"

main()
{
  IS is;
  int total;
  int i, j;

  is = new_inputstruct(NULL);
  total = 0;

  while (get_line(is) >= 0) {
    for (i = 0; i < is->NF; i++) {
      if (sscanf(is->fields[i], "%d", &j) == 1) {
        total += ((i+1)*j);
      } else {
        fprintf(stderr, "Line %d field %d\n", 
                is->line, i);
        exit(1);
      }
    }
  }
  printf("%d\n", total);
}
