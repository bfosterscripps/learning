CS140 -- Midterm Exam: Q4 - Q7

The following code is for questions 4 through 7 
char *revstring(char *s)
{
  int i, len;
  char tmp;

  len = strlen(s);

  for (i = 0; i < len/2; i++) {
    tmp = s[i];
    s[i] = s[len-i-1];
    s[len-i-1] = tmp;
  }
  return s;
}
